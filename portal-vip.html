async function verificarProtocolo() { const protocolo = document.getElementById('input-protocolo').value.trim().toUpperCase(); const btn = document.getElementById('btn-acesso'); const msgErro = document.getElementById('msg-erro'); if(!protocolo) return; msgErro.style.display = "none"; btn.innerText = "LOCALIZANDO..."; try { // AGORA SIM: O Motor busca a informação real no Satélite Cloudflare const req = await fetch(API_MATRIZ + "?protocolo=" + protocolo); if (!req.ok) { throw new Error("Protocolo não encontrado ou ainda em análise."); } const dados = await req.json(); // Abre a sala segura com o LAUDO REAL que você escreveu no Bunker abrirSalaSegura(dados.laudo); } catch (e) { btn.innerText = "Descriptografar Laudo"; msgErro.style.display = "block"; } }
